def _deep_fill(obj, defaults):
    """閫掑綊濉厖缂哄け鐨勯敭锛屼笉瑕嗙洊宸叉湁鍊?""

    for k, v in obj.items():
        if k in defaults:

        else:
            result[k] = v
    # 琛ュ厖 obj 缂哄け浣?defaults 鏈夌殑閿?    for k, v in defaults.items():

import yaml
from pathlib import Path
from copy import deepcopy

class EquivalenceRules:
    """鍔犺浇鍜屽簲鐢ㄥ瓧娈电瓑鏁堣鍒欙紙鍗曚緥锛?""

    _instance = None
    _rules: dict = {}

    def __new__(cls):
        if cls._instance is None:
            cls._instance = super().__new__(cls)
            cls._instance._load()
        return cls._instance

    def _load(self):
        path = Path(__file__).parent.parent / "config" / "equivalence_rules.yaml"
        if path.exists():
            with open(path) as f:
                self._rules = yaml.safe_load(f) or {}
        else:
            self._rules = {}

    def get_field_default(self, resource_type: str, field: str):
        return self._rules.get(resource_type, {}).get("field_defaults", {}).get(field)

    def get_json_field_rules(self, resource_type: str, field: str):
        """鑾峰彇 JSON 瀛楁鐨勫～鍏呰鍒?""
        return self._rules.get(resource_type, {}).get("json_fields", {}).get(field, {})

    def should_ignore_edge_field(self, resource_type: str, field: str) -> bool:
        return field in self._rules.get(resource_type, {}).get("ignore_edge_fields", [])

    def get_field_alias(self, resource_type: str, field: str) -> str:
        return self._rules.get(resource_type, {}).get("field_aliases", {}).get(field, field)
